Disk Scheduling Program



class FCFS_DiskScheduling:
    def __init__(self, requests, head_start, disk_size):
        self.requests = requests
        self.head_start = head_start
        self.disk_size = disk_size
        self.size = len(requests)

    def calculate_seek_count(self):
        if self.head_start not in self.requests:
            print("Invalid head position. Please enter a valid head position.")
            return -1

        seek_count = 0
        head = self.head_start

        for i in range(self.size):
            cur_track = self.requests[i]
            distance = abs(cur_track - head)
            seek_count += distance
            head = cur_track

        return seek_count

    def print_results(self):
        seek_count = self.calculate_seek_count()
        if seek_count != -1:
            print("Total number of seek operations =", seek_count)
            print("Seek Sequence is:")
            for i in range(self.size):
                print(self.requests[i])

    fcfs_scheduler = FCFS_DiskScheduling(arr, head, disk_size)
